## Usage

```js
var Prompt = require('{%= name %}');
var prompt = new Prompt({
  type: 'sort',
  name: 'colors',
  message: 'Put the colors in order of most preferred to least.',
  choices: [
    'black',
    'blue',
    'green',
    'red',
    'white',
    'yellow'
  ]
});

prompt.run()
  .then(function(answer) {
    console.log(answer);
  });

// or
prompt.ask(function(answer) {
  console.log(answer);
});

```
